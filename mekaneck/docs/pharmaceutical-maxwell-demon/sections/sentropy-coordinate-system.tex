\subsection{Tri-Dimensional Categorical Space}

Pharmaceutical navigation occurs in three-dimensional S-entropy space encoding information deficit, temporal distance, and configurational entropy:

\begin{definition}[S-Entropy Coordinates]
For oscillatory frequency $\omega$, the S-entropy coordinate triple is:
\begin{align}
S_{\text{knowledge}} &= -\log_2 P_{\text{config}}(\omega) = \log_2(|\Omega_{\omega}|) \label{eq:sk} \\
S_{\text{time}} &= \log_{10}(\tau / \tau_0) = -\log_{10}(\omega / \omega_0) \label{eq:st} \\
S_{\text{entropy}} &= -\sum_{i=1}^{N_{\text{phase}}} p_i \log_2 p_i \label{eq:se}
\end{align}
where $P_{\text{config}}$ is configuration probability, $\tau = 2\pi/\omega$ is characteristic timescale, $\tau_0 = 1$ s reference, and $p_i$ is phase distribution probability.
\end{definition}

\textbf{Interpretation}:
\begin{itemize}
\item $S_k$: Information needed to specify configuration (bits)
\item $S_t$: Log-temporal distance to equilibrium (orders of magnitude)
\item $S_e$: Shannon entropy of phase distribution (bits)
\end{itemize}

\subsection{Categorical Distance Metric}

\begin{definition}[S-Entropy Distance]
The categorical distance between two points $\mathbf{s}_1 = (s_k^1, s_t^1, s_e^1)$ and $\mathbf{s}_2$ is:
\begin{equation}
d_{\text{cat}}(\mathbf{s}_1, \mathbf{s}_2) = \sqrt{\alpha_k (s_k^1 - s_k^2)^2 + \alpha_t (s_t^1 - s_t^2)^2 + \alpha_e (s_e^1 - s_e^2)^2}
\end{equation}
with dimensional weights $\alpha_k = 1, \alpha_t = 0.5, \alpha_e = 1$ reflecting relative importance.
\end{definition}

\textbf{Metric Properties}: Validated triangle inequality $d_{13} \leq d_{12} + d_{23}$ for 1,000 random triplets (100\% satisfaction).
